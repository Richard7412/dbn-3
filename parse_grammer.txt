[DBN_PROGRAM] := ([DBN_PROGRAM_STATEMENT] | NEWLINE)*

[DBN_PROGRAM_STATEMENT] := [DBN_DEFINE_COMMAND] |
                           [DBN_LOAD] |
                           [DBN_STATEMENT]

[DBN_DEFINE_COMMAND] := COMMAND WORD (WORD)* [DBN_STATEMENT_BLOCK]

[DBN_LOAD] := LOAD PATH

[DBN_STATEMENT] := [DBN_SET] |
                   [DBN_REPEAT] |
                   [DBN_COMMAND] |
                   [DBN_QUESTION] | 
                   [DBN_COMMENT]

[DBN_STATEMENT_BLOCK] := NEWLINE* OPENBRACE NEWLINE [DBN_BODY] CLOSE_BRACE NEWLINE

[DBN_BODY] := ([DBN_STATEMENT] | NEWLINE)*
                   
[DBN_SET] := SET ([DBN_DOT] | WORD) [DBN_ARGUMENT] NEWLINE

[DBN_REPEAT] := REPEAT WORD [DBN_ARGUMENT] [DBN_ARGUMENT] [DBN_STATEMENT_BLOCK]

[DBN_QUESTION] := [DBN_QUESTION_WORD] [DBN_ARGUMENT] [DBN_ARGUMENT] [DBN_STATEMENT_BLOCK]

[DBN_QUESTION_WORD] := SMALLER | NOTSMALLER | SAME | NOTSAME

[DBN_COMMAND] := WORD [DBN_ARGUMENT]* NEWLINE

[DBN_DOT] := OPENBRACKET [DBN_ARGUMENT] [DBN_ARGUMENT] CLOSEBRACKET

[DBN_ARGUMENT] := NUMBER |
                  WORD |
                  [DBN_DOT] |
                  [DBN_ARITHMETIC]

[DBN_ARITHMETIC] := OPENPAREN [DBN_ARITHMETIC_ARGUMENT] CLOSEPAREN

[DBN_ARITHMETIC_ARGUMENT] := [DBN_ARGUMENT]
                             [DBN_BINARY_OPERATION]

[DBN_BINARY_OPERATION] := [DBN_ARITHMETIC_ARGUMENT] OPERATION [DBN_ARITHMETIC_ARGUMENT]
